499 void flush(SC16IS752_Channel channel) {
500 if (channel > SC16IS752_CHANNEL_B) {
501 return;
502 }
503
504 uint8_t tmp_lsr;
505 uint32_t start = millis();
506
507 do {
508 tmp_lsr = readRegister(channel, SC16IS750_REG_LSR);
509 if ((tmp_lsr & 0x20) != 0) {
510 return;
511 }
512 } while ((millis() - start) < timeout);
513
514 if (logger) {
515 logger->warn("flush timeout ch=%d", channel);
516 }
517 }

525 int interruptEventTest(SC16IS752_Channel channel) {
526 uint8_t irq_src;
527
528 irq_src = readRegister(channel, SC16IS750_REG_IIR);
529 // irq_src = (irq_src >> 1);
530 // irq_src &= 0x3F;
531 irq_src &= 0x3E;
532
533 switch (irq_src) {
534 case 0x06: // Receiver Line Status Error
535 return SC16IS750_RECEIVE_LINE_STATUS_ERROR;
536 case 0x0c: // Receiver time-out interrupt
537 return SC16IS750_RECEIVE_TIMEOUT_INTERRUPT;
538 case 0x04: // RHR interrupt
539 return SC16IS750_RHR_INTERRUPT;
540 case 0x02: // THR interrupt
541 return SC16IS750_THR_INTERRUPT;
542 case 0x00: // modem interrupt;
543 return SC16IS750_MODEM_INTERRUPT;
544 case 0x30: // input pin change of state
545 return SC16IS750_INPUT_PIN_CHANGE_STATE;
546 case 0x10: // XOFF
547 return SC16IS750_RECEIVE_XOFF;
548 case 0x20: // CTS,RTS
549 return SC16IS750_CTS_RTS_CHANGE;
550 default:
551 return -1;
552 }
553 return -1;
554 }

660 int16_t setBaudrate(
661 SC16IS752_Channel channel,
662 uint32_t baudrate) { // return error of baudrate parts per thousand
663 uint16_t divisor;
664 uint8_t prescaler;
665 uint32_t actual_baudrate;
666 int16_t error;
667 uint8_t temp_lcr;
668 if ((readRegister(channel, SC16IS750_REG_MCR) & 0x80) ==
669 0) { // if prescaler==1
670 prescaler = 1;
671 } else {
672 prescaler = 4;
673 }
674
675 divisor = (crystal_freq / prescaler) / (baudrate * 16);
676
677 temp_lcr = readRegister(channel, SC16IS750_REG_LCR);
678 temp_lcr |= 0x80;
679 writeRegister(channel, SC16IS750_REG_LCR, temp_lcr);
680 // write to DLL
681 writeRegister(channel, SC16IS750_REG_DLL, (uint8_t)divisor);
682 // write to DLH
683 writeRegister(channel, SC16IS750_REG_DLH, (uint8_t)(divisor >> 8));
684 temp_lcr &= 0x7F;
685 writeRegister(channel, SC16IS750_REG_LCR, temp_lcr);
686
687 actual_baudrate = (crystal_freq / prescaler) / (16 * divisor);
688 error = ((float)actual_baudrate - baudrate) * 1000 / baudrate;
689 if (logger) {
690 logger->debug("Desired baudrate: %d", baudrate);
691 logger->debug("Calculated divisor: %d", divisor);
692 logger->debug("Actual baudrate: %d", actual_baudrate);
693 logger->debug("Baudrate error: %d", (int)error);
694 }
695
696 return error;
697 }

699 uint8_t readRegister(SC16IS752_Channel channel, uint8_t reg_addr) {
700 uint8_t result;
701 if (wire_bus != nullptr) { // register read operation via I2C
702 wire_bus->beginTransmission(device_address_sspin);
703 wire_bus->write((reg_addr << 3 | channel << 1));
704 wire_bus->endTransmission(0);
705 wire_bus->requestFrom(device_address_sspin, (uint8_t)1);
706 result = wire_bus->read();
707 } else if (spi_bus != nullptr) { // register read operation via SPI
708 spi_bus->beginTransaction(spi_settings);
709 ::digitalWrite(device_address_sspin, LOW);
710 delayMicroseconds(10);
711 spi_bus->transfer(0x80 | ((reg_addr << 3 | channel << 1)));
712 result = spi_bus->transfer(0xff);
713 delayMicroseconds(10);
714 ::digitalWrite(device_address_sspin, HIGH);
715 spi_bus->endTransaction();
716 } else {
717 result = 0;
718 }
719 if (logger) {
720 logger->debug("ReadRegister ch=%X reg=%X res=%X", channel,
721 (reg_addr << 3 | channel << 1), result);
722 }
723 return result;
724 }

726 void writeRegister(SC16IS752_Channel channel, uint8_t reg_addr, uint8_t val) {
727 if (logger) {
728 logger->debug("WriteRegister ch=%X reg=%X val=%X", channel,
729 (reg_addr << 3 | channel << 1), val);
730 }
731
732 if (wire_bus != nullptr) { // register write operation via I2C
733 wire_bus->beginTransmission(device_address_sspin);
734 wire_bus->write((reg_addr << 3 | channel << 1));
735 wire_bus->write(val);
736 wire_bus->endTransmission(1);
737 } else { // register write operation via SPI
738 spi_bus->beginTransaction(spi_settings);
739 ::digitalWrite(device_address_sspin, LOW);
740 delayMicroseconds(10);
741 spi_bus->transfer((reg_addr << 3 | channel << 1));
742 spi_bus->transfer(val);
743 delayMicroseconds(10);
744 ::digitalWrite(device_address_sspin, HIGH);
745 spi_bus->endTransaction();
746 }
747 return;
748 }

750 void setLine(SC16IS752_Channel channel, uint8_t data_length,
751 uint8_t parity_select, uint8_t stop_length) {
752 uint8_t temp_lcr;
753 temp_lcr = readRegister(channel, SC16IS750_REG_LCR);
754 temp_lcr &= 0xC0; // Clear the lower six bit of LCR (LCR[0] to LCR[5]
755 if (logger) {
756 logger->debug("LCR Register: %d", temp_lcr);
757 }
758 switch (data_length) { // data length settings
759 case 5:
760 break;
761 case 6:
762 temp_lcr |= 0x01;
763 break;
764 case 7:
765 temp_lcr |= 0x02;
766 break;
767 case 8:
768 temp_lcr |= 0x03;
769 break;
770 default:
771 temp_lcr |= 0x03;
772 break;
773 }
774
775 if (stop_length == 2) {
776 temp_lcr |= 0x04;
777 }
778
779 // parity selection length settings
780 switch (parity_select) {
781 case 0: // no parity
782 break;
783 case 1: // odd parity
784 temp_lcr |= 0x08;
785 break;
786 case 2: // even parity
787 temp_lcr |= 0x18;
788 break;
789 case 3: // force '1' parity
790 temp_lcr |= 0x03;
791 break;
792 case 4: // force '0' parity
793 break;
794 default:
795 break;
796 }
797
798 writeRegister(channel, SC16IS750_REG_LCR, temp_lcr);
799 }

915 void fifoSetTriggerLevel(SC16IS752_Channel channel, uint8_t rx_fifo,
916 uint8_t length) {

919 temp_reg = readRegister(channel, SC16IS750_REG_MCR);
920 temp_reg |= 0x04;
921 writeRegister(channel, SC16IS750_REG_MCR,
922 temp_reg); // SET MCR[2] to '1' to use TLR register or
923 // trigger level control in FCR register
924

926 writeRegister(
927 channel, SC16IS750_REG_EFR,
928 temp_reg | 0x10); // set ERF[4] to '1' to use the enhanced features
929 if (rx_fifo == 0) {
930 writeRegister(channel, SC16IS750_REG_TLR,
931 length << 4); // Tx FIFO trigger level setting
932 } else {
933 writeRegister(channel, SC16IS750_REG_TLR,
934 length); // Rx FIFO Trigger level setting
935 }
936 writeRegister(channel, SC16IS750_REG_EFR,
937 temp_reg); // restore EFR register
938
939 return;
